```java
// 实现 loadFromApplicationConfig
private List<McpServerConfig> loadFromApplicationConfig() {
    List<McpServerConfig> configs = new ArrayList<>();
    
    // 方法1: 使用 @ConfigurationProperties
    // 方法2: 手动解析 Environment 的 mcp.servers.*
    
    String[] serverNames = environment.getProperty("app.mcp.servers", String[].class, new String[0]);
    for (String name : serverNames) {
        String url = environment.getProperty("app.mcp.servers." + name + ".url");
        if (url != null) {
            configs.add(new McpServerConfig(
                name, 
                McpConfigScope.PROJECT,
                null,  // command
                List.of(),  // args
                url,   // SSE URL
                Map.of(),  // env
                null   // apiKey
            ));
        }
    }
    return configs;
}
```
