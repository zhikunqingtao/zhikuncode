public sealed interface MessageParam {

    record User(List<ContentPart> parts) implements MessageParam {}
    record Assistant(String text, List<ToolCall> toolCalls) implements MessageParam {}

## 四、Anthropic 原生 Provider

**文件**: 新建 `backend/src/main/java/com/aicodeassistant/llm/impl/AnthropicProvider.java`

### 4.1 与 OpenAI 兼容模式的差异

| 维度 | OpenAI Compatible | Anthropic Native | 影响 |
|------|-------------------|-------------------|------|
| API 端点 | `/v1/chat/completions` | `/v1/messages` | URL 不同 |
| system 消息 | 放在 messages 数组内 | 独立 `system` 字段 | 消息结构不同 |
| tool_use 响应 | `tool_calls` in delta | `content_block` type=`tool_use` | 流解析逻辑不同 |
| cache_control | 不支持 | `cache_control:{"type":"ephemeral"}` | 费用节省 50-90% |
| thinking | 不支持 | `thinking:{"type":"enabled","budget_tokens":N}` | 推理能力解锁 |
| 认证头 | `Authorization: Bearer` | `x-api-key` + `anthropic-version` | Header 不同 |
| 流格式 | `data: {choices:[{delta:...}]}` | `event: content_block_delta\ndata: {...}` | SSE 解析不同 |

### 4.2 实现要点

```java
@Service
@ConditionalOnProperty(name = "llm.provider", havingValue = "anthropic")
public class AnthropicProvider implements LlmProvider {

    // 1. 请求构建
    private ObjectNode buildRequest(String model, List<MessageParam> messages,
                                     String systemPrompt, List<ToolDefinition> tools,
                                     int maxTokens, ThinkingConfig thinking) {
        ObjectNode root = objectMapper.createObjectNode();
        root.put("model", model);
        root.put("max_tokens", maxTokens);
        root.put("stream", true);

        // system 独立字段 + cache_control
        ArrayNode systemArray = root.putArray("system");
        ObjectNode sysPart = systemArray.addObject();
        sysPart.put("type", "text");
        sysPart.put("text", systemPrompt);
        sysPart.putObject("cache_control").put("type", "ephemeral");

        // thinking 参数
        if (thinking instanceof ThinkingConfig.Enabled enabled) {
            ObjectNode thinkNode = root.putObject("thinking");
            thinkNode.put("type", "enabled");
            thinkNode.put("budget_tokens", enabled.budgetTokens());
        }

        // messages + tools 转换...
        return root;
    }

    // 2. SSE 解析 — Anthropic 特有事件类型
    // event: message_start     → 初始化
    // event: content_block_start → 新内容块开始（text/tool_use/thinking）
    // event: content_block_delta → 增量内容
    // event: content_block_stop  → 内容块结束
    // event: message_delta      → 最终 usage + stop_reason
    // event: message_stop       → 流结束
}
```

// 模糊搜索算法
function fuzzyMatch(query: string, items: CommandItem[]): CommandItem[] {
    return items.filter(item =>
        item.name.toLowerCase().includes(query.toLowerCase())
    ).sort((a, b) => {
        // 前缀匹配优先
        const aStarts = a.name.startsWith(query) ? 0 : 1;

        return aStarts - bStarts;
    });

### 7.2 ripgrep 后端改造

```java
private ToolResult callWithRipgrep(String pattern, String searchPath,
                                     boolean caseInsensitive, List<String> include) {
    List<String> cmd = new ArrayList<>(List.of("rg", "--json", "--max-count", "50"));
    if (caseInsensitive) cmd.add("-i");
    for (String glob : include) {
        cmd.add("--glob"); cmd.add(glob);
    }
    cmd.add(pattern);
    cmd.add(searchPath);

    ProcessBuilder pb = new ProcessBuilder(cmd);
    pb.redirectErrorStream(true);
    Process proc = pb.start();
    String output = new String(proc.getInputStream().readAllBytes(), StandardCharsets.UTF_8);
    int exitCode = proc.waitFor();

    if (exitCode == 0) {
        return ToolResult.success(formatRipgrepJson(output));
    } else if (exitCode == 1) {
        return ToolResult.success("No matches found");
    } else {
        // ripgrep 不可用，降级
        return callWithJavaNio(pattern, searchPath, caseInsensitive, include);
    }
}

private boolean isRipgrepAvailable() {
    try {
        Process p = Runtime.getRuntime().exec(new String[]{"rg", "--version"});
        return p.waitFor() == 0;
    } catch (Exception e) {
        return false;
    }
}
```

### 8.1 过滤规则

```java
@Component
public class SensitiveDataFilter {

    private static final List<Pattern> PATTERNS = List.of(
        // API Keys
        Pattern.compile("(sk-[a-zA-Z0-9]{20,})"),                    // OpenAI
        Pattern.compile("(AKIA[0-9A-Z]{16})"),                       // AWS
        Pattern.compile("(ghp_[a-zA-Z0-9]{36})"),                    // GitHub
        // 通用 key=value
        Pattern.compile("(?i)(api[_-]?key|secret|password|token|auth)\\s*[=:]\\s*['\"]?([^\\s'\"]{8,})"),
        // JWT
        Pattern.compile("(eyJ[a-zA-Z0-9_-]{10,}\\.[a-zA-Z0-9_-]{10,}\\.[a-zA-Z0-9_-]{10,})"),
        // 私钥
        Pattern.compile("-----BEGIN (RSA |EC |DSA )?PRIVATE KEY-----")
    );

    public String filter(String content) {
        if (content == null) return null;
        String result = content;
        for (Pattern p : PATTERNS) {
            result = p.matcher(result).replaceAll("***REDACTED***");
        }
        return result;
    }
}
```
